1%% Increase waypoint2
 2% waypoint = [22.0000,120.0009;22.0001,120.0007];
 3function [waypoint_add_lat,waypoint_add_lon,waypoint_add_amount] = increase_waypoint2(lat,lon)
 4% lat = [22.0001;22.0004];
 5% lon = [120.0001;120.0004];
 6
 7step = round(abs(lon(2) - lon(1))*10000);
 8waypoint_add_amount = step+1;
 9waypoint_add_lat = zeros(waypoint_add_amount,1);
10waypoint_add_lon = zeros(waypoint_add_amount,1);
11slope = abs((lat(2)-lat(1))/(lon(2)-lon(1)));
12for j = 0:step-1
13    waypoint_add_lon(j+1) = lon(1) + 0.0001*j*sign(lon(2)-lon(1));
14    waypoint_add_lat(j+1) = lat(1) + slope*0.0001*j*sign(lat(2)-lat(1));
15end
16waypoint_add_lat(waypoint_add_amount) = lat(2);
17waypoint_add_lon(waypoint_add_amount) = lon(2);
18end
