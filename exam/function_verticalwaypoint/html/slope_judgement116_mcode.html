1%% slope judgement
 2function vertical_waypoint = slope_judgement(H,waypoint_add,slope,x_coordination,num_point,slope_threshold)
 3vertical_waypoint = zeros(num_point,3);
 4
 5point_need_change = find(slope > slope_threshold);
 6ps = 0;
 7pf = 0;
 8for i = 1:size(point_need_change,1)
 9    point_advance = 0;
10    param = slope(point_need_change(i,1));
11    while (param > slope_threshold)
12        point_advance = point_advance + 1;
13        ps = point_need_change(i,1) - point_advance;
14        pf = point_need_change(i,1) + 1;
15%         if ps == 0
16%             error;
17%         end
18        param = HIGHT2SLOPE([H(ps);H(pf)],x_coordination(pf)-x_coordination(ps));
19    end
20    ps_next = ps+1;
21    pf_prev = pf-1;
22    hight_change = H(ps) + param*(x_coordination(ps_next:pf_prev)-x_coordination(ps));
23    H(ps+1:pf-1) = hight_change;
24    %record_same_slope(i,1:point_advance) = ps+1:pf-1;
25end
26vertical_waypoint(:,1:2) = waypoint_add;
27vertical_waypoint(:,3) = H;
28end
