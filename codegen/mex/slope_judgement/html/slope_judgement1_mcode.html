 3function [vertical_waypoint,unvalid] = slope_judgement(H,slope,x_coordination,num_point,slope_threshold)
 4vertical_waypoint = zeros(num_point,1);
 5unvalid = 0;
 6point_need_change = find(slope > slope_threshold);
 7ps = 0;
 8pf = 0;
 9for i = 1:size(point_need_change,1)
10    point_advance = 0;
11    param = slope(point_need_change(i,1));
12    while (param > slope_threshold)
13        point_advance = point_advance + 1;
14        ps = point_need_change(i,1) - point_advance;
15        pf = point_need_change(i,1) + 1;
16        if ps <= 0
17            unvalid = 1;
18            ps = 1;
19            break;
20        end
21        param = HIGHT2SLOPE([H(ps);H(pf)],x_coordination(pf)-x_coordination(ps));
22    end
23    ps_next = ps+1;
24    pf_prev = pf-1;
25    hight_change = H(ps) + param*(x_coordination(ps_next:pf_prev)-x_coordination(ps));
26    H(ps+1:pf-1) = hight_change;
27    %record_same_slope(i,1:point_advance) = ps+1:pf-1;
28end
29vertical_waypoint(:,1) = H;
30end
