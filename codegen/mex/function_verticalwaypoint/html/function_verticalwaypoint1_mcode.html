1function [vertical_waypoint,index_of_waypoint] = function_verticalwaypoint(waypoint,map)
 2% map = imread('ASTGTM2_N24E121_dem.tif');
 3% waypoint = [24.92021,121.70231;
 4%             24.70202,121.90121;
 5%             24.90202,121.70122];
 6temp_waypoint = floor(waypoint*10000);
 7waypoint = temp_waypoint/10000;
 8%% Increase waypoint
 9step_accu = 0;
10accu = 0;
11for i = 1:size(waypoint,1)-1
12    steps = floor(abs(waypoint(i+1,2) - waypoint(i,2))*10000);
13    accu = steps + accu;
14end
15waypoint_add = zeros(accu+1,2);
16for i = 1:size(waypoint,1)-1
17    dis_latlon = waypoint(i+1,:) - waypoint(i,:);
18    slope = abs(dis_latlon(1)/dis_latlon(2));
19    steps = abs(round(dis_latlon(2)/0.0001));
20    for j = 0:steps-1
21        waypoint_add(j+1+step_accu,2) = waypoint(i,2) + 0.0001*j*sign(dis_latlon(2));
22        waypoint_add(j+1+step_accu,1) = waypoint(i,1) + slope*0.0001*j*sign(dis_latlon(1));
23    end
24    step_accu = steps + step_accu;
25end
26waypoint_add(1+step_accu,:) = waypoint(end,:);
27
28%% Calculate distance
29InitLat = waypoint(1,1);
30dis_accu = zeros(accu+1,1);
31dis_meter = zeros(accu,1);
32MetperLat = 110574;
33MetperLon = 111320*cos(InitLat*pi/180); % meter per degree
34for i = 1:size(waypoint_add,1)-1
35    dis_latlon = waypoint_add(i+1,:) - waypoint_add(i,:);
36    dis_meter(i,1) = sqrt((dis_latlon(1)*MetperLat)^2  + (dis_latlon(2)*MetperLon)^2);
37    dis_accu(i+1,1) = dis_accu(i,1) + dis_meter(i,1);
38end
39
40%% Calculate slope
41[H,index_of_waypoint] = GPS2HIGHTC(waypoint_add(:,1),waypoint_add(:,2),map);
42slope = HIGHT2SLOPE(double(H),dis_meter);
43
44
45%% slope judgement
46hight2 = double(H);
47slope_threshold = 0.15;
48
49point_need_change = find(slope > slope_threshold);
50ps = 0;
51pf = 0;
52for i = 1:size(point_need_change,1)
53    point_advance = 0;
54    param = slope(point_need_change(i,1));
55    while (param > slope_threshold)
56        point_advance = point_advance + 1;
57        ps = point_need_change(i,1) - point_advance;
58        pf = point_need_change(i,1) + 1;
59        param = HIGHT2SLOPE([hight2(ps);hight2(pf)],dis_accu(pf)-dis_accu(ps));
60    end
61    ps_next = ps+1;
62    pf_prev = pf-1;
63    hight_change = hight2(ps) + param*(dis_accu(ps_next:pf_prev)-dis_accu(ps));
64    hight2(ps+1:pf-1) = hight_change;
65    %record_same_slope(i,1:point_advance) = ps+1:pf-1;
66end
67vertical_waypoint = zeros(accu+1,3);
68vertical_waypoint(:,1:2) = waypoint_add;
69vertical_waypoint(:,3) = hight2;
70%% delete redundant point
71delete_matrix = ones(size(H,1),1);
72slope2 = HIGHT2SLOPE(hight2,dis_meter);
73for i = 1:size(slope2,1)-1
74    sp = i;
75    repeat = 1;
76    while slope2(sp) == slope2(sp+repeat)
77        repeat = repeat + 1;
78    end
79    fp = sp + repeat;
80    if repeat > 1
81        delete_matrix(sp+1:fp-1) = 0;
82    end
83end
84index = find(delete_matrix == 0);
85% dis_accu(index) = [];
86index_of_waypoint(index,:) = [];
87vertical_waypoint(index,:) = [];
88end
