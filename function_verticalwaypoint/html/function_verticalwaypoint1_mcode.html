1function [vertical_waypoint,index_of_waypoint,dis_accu] = function_verticalwaypoint(waypoint,map)
 2% map = imread('ASTGTM2_N24E121_dem.tif');
 3% waypoint = [24.92021,121.70231;
 4%             24.70202,121.90121;
 5%             24.90202,121.70122];
 6% map = imread('ASTGTM2_N25E055_dem.tif');
 7% waypoint = [25.0783,55.4982;
 8%             25.0883,55.5082;
 9%             25.0983,55.5182;
10%             25.4983,55.7182];
11temp_waypoint = floor(waypoint*10000);
12waypoint = temp_waypoint/10000;
13%% Increase waypoint
14step_accu = 0;
15accu = 0;
16for i = 1:size(waypoint,1)-1
17    steps1 = floor(abs(waypoint(i+1,2) - waypoint(i,2))*10000);
18    accu = steps1 + accu;
19end
20waypoint_add = zeros(accu+1,2);
21for i = 1:size(waypoint,1)-1
22    dis_latlon = waypoint(i+1,:) - waypoint(i,:);
23    slope = abs(dis_latlon(1)/dis_latlon(2));
24    steps = abs(floor(dis_latlon(2)*10000));
25    for j = 0:steps-1
26        waypoint_add(j+1+step_accu,2) = waypoint(i,2) + 0.0001*j*sign(dis_latlon(2));
27        waypoint_add(j+1+step_accu,1) = waypoint(i,1) + slope*0.0001*j*sign(dis_latlon(1));
28    end
29    step_accu = steps + step_accu;
30end
31waypoint_add(1+step_accu,:) = waypoint(end,:);
32
33%% Calculate distance
34InitLat = waypoint(1,1);
35dis_accu = zeros(accu+1,1);
36dis_meter = zeros(accu,1);
37MetperLat = 110574;
38MetperLon = 111320*cos(InitLat*pi/180); % meter per degree
39for i = 1:size(waypoint_add,1)-1
40    dis_latlon = waypoint_add(i+1,:) - waypoint_add(i,:);
41    dis_meter(i,1) = sqrt((dis_latlon(1)*MetperLat)^2  + (dis_latlon(2)*MetperLon)^2);
42    dis_accu(i+1,1) = dis_accu(i,1) + dis_meter(i,1);
43end
44
45%% Calculate slope
46[H,index_of_waypoint] = GPS2HIGHTC(waypoint_add(:,1),waypoint_add(:,2),map);
47slope = HIGHT2SLOPE(double(H),dis_meter);
48
49
50%% slope judgement
51hight2 = double(H);
52slope_threshold = 0.15;
53
54point_need_change = find(slope > slope_threshold);
55ps = 0;
56pf = 0;
57for i = 1:size(point_need_change,1)
58    point_advance = 0;
59    param = slope(point_need_change(i,1));
60    while (param > slope_threshold)
61        point_advance = point_advance + 1;
62        ps = point_need_change(i,1) - point_advance;
63        pf = point_need_change(i,1) + 1;
64%         if ps == 0
65%             error;
66%         end
67        param = HIGHT2SLOPE([hight2(ps);hight2(pf)],dis_accu(pf)-dis_accu(ps));
68    end
69    ps_next = ps+1;
70    pf_prev = pf-1;
71    hight_change = hight2(ps) + param*(dis_accu(ps_next:pf_prev)-dis_accu(ps));
72    hight2(ps+1:pf-1) = hight_change;
73    %record_same_slope(i,1:point_advance) = ps+1:pf-1;
74end
75vertical_waypoint = zeros(accu+1,3);
76vertical_waypoint(:,1:2) = waypoint_add;
77vertical_waypoint(:,3) = hight2;
78%% delete redundant point
79delete_matrix = ones(size(H,1),1);
80slope2 = HIGHT2SLOPE(hight2,dis_meter);
81for i = 1:size(slope2,1)-1
82    sp = i;
83    repeat = 1;
84    while slope2(sp) == slope2(sp+repeat)
85        repeat = repeat + 1;
86    end
87    fp = sp + repeat;
88    if repeat > 1
89        delete_matrix(sp+1:fp-1) = 0;
90    end
91end
92index = find(delete_matrix == 0);
93dis_accu(index) = [];
94index_of_waypoint(index,:) = [];
95vertical_waypoint(index,:) = [];
96end
