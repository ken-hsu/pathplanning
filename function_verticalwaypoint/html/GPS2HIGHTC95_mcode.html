1%% GPS to HIGHT
 2% A function with a given DEM, input a GPS location, output a terrian
 3% Input double(Longitude,Altitude)
 4% Output int16(Hight)
 5function [H,c_record] = GPS2HIGHTC(Latitude,Lontitude,map)
 6% map = imread('ASTGTM2_N24E121_dem.tif');
 7%     Latitude = 24.0001;
 8%     Lontitude = 120.0001;
 9    MapSize = 3601;
10    amount = size(Latitude,1);
11    c_record = zeros(amount,2);
12    H = zeros(amount,1);
13    for i = 1:amount
14        Lat = Latitude(i);
15        Lon = Lontitude(i);
16        LonDec = (Lon - floor(Lon)); % take the number after the decimal point
17        LonDec4 = floor(LonDec*10000); % the number of 4 variable after the decimal point
18        LonDec4p1 = round(LonDec4*0.36);
19        if LonDec4*0.36 - LonDec4p1 > 0
20            flagLon = 1;
21        else
22            flagLon = -1;
23        end
24        LonDecCollumn = LonDec4p1+1;
25
26
27        LatDec = (Lat - floor(Lat));
28        LatDec4 = floor(LatDec*10000);
29        LatDec4p1 = round(LatDec4*0.36);
30        if LatDec4*0.36 - LatDec4p1 > 0
31            flagLat = -1;
32        else
33            flagLat = 1;
34        end
35        LatDecRow = (MapSize+1) - (LatDec4p1+1); % Index of latitude accounted from lower row to higher
36        c = [LatDecRow,LonDecCollumn];
37        c1 = [LatDecRow+flagLat,LonDecCollumn];
38        c2 = [LatDecRow,LonDecCollumn+flagLon];
39        if c1(1) > 3601 || c1(1) < 1
40            c1(1) = c(1);
41        end
42        if c2(2) > 3601 || c2(2) < 1
43            c2(2) = c(2);
44        end
45%         H = map(c(1),c(2));
46    %     H = (map(c(1),c(2))+map(c1(1),c1(2))+map(c2(1),c2(2)))/3;
47    c_record(i,:) = c;
48          H(i,1) = max([map(c(1),c(2)), map(c(1),c(2)) ,map(c2(1),c2(2)) ]);
49    end
50end
